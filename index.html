function doGet() { const DATABASE_ID = "314f89d90a6380a388abd1fe7cb3277f"; const NOTION_TOKEN = "あなたのNotionトークン"; const url = "https://api.notion.com/v1/databases/" + DATABASE_ID + "/query"; const options = { method: "post", headers: { "Authorization": "Bearer " + NOTION_TOKEN, "Notion-Version": "2022-06-28", "Content-Type": "application/json" } }; const response = UrlFetchApp.fetch(url, options); const data = JSON.parse(response.getContentText()); const phrases = data.results.map(page => { let phrase = ""; let book = ""; let author = ""; let cover = ""; /* フレーズ */ if( page.properties["フレーズ"] && page.properties["フレーズ"].title ){ phrase = page.properties["フレーズ"].title .map(t => t.plain_text) .join(""); } /* 本のタイトル */ if( page.properties["本のタイトル"] && page.properties["本のタイトル"].select ){ book = page.properties["本のタイトル"].select.name; } /* 著者 */ if( page.properties["著者"] && page.properties["著者"].select ){ author = page.properties["著者"].select.name; } /* 書影 */ if( page.properties["書影"] && page.properties["書影"].files && page.properties["書影"].files.length > 0 ){ const file = page.properties["書影"].files[0]; if(file.external){ cover = file.external.url; } if(file.file){ cover = file.file.url; } } return { phrase: phrase, book: book, author: author, cover: cover, notion: page.url }; }); return ContentService .createTextOutput(JSON.stringify(phrases)) .setMimeType(ContentService.MimeType.JSON); }
