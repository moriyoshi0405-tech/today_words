function doGet() { const DATABASE_ID = "314f89d90a6380a388abd1fe7cb3277f"; const NOTION_TOKEN = "あなたのNotionトークン"; const res = UrlFetchApp.fetch( `https://api.notion.com/v1/databases/${DATABASE_ID}/query`, { method: "post", headers: { "Authorization": "Bearer " + NOTION_TOKEN, "Notion-Version": "2022-06-28", "Content-Type": "application/json" } } ); const data = JSON.parse(res.getContentText()); const phrases = data.results.map(p => ({ phrase: p.properties["フレーズ"]?.title?.map(t => t.plain_text).join("") || "", book: p.properties["本のタイトル"]?.select?.name || "", author: p.properties["著者"]?.select?.name || "", cover: p.properties["書影"]?.files?.[0]?.file?.url || p.properties["書影"]?.files?.[0]?.external?.url || "", notion: p.url })); return ContentService.createTextOutput(JSON.stringify(phrases)) .setMimeType(ContentService.MimeType.JSON); }
